% Creates a new command to include a Bash script, the first parameter is the filename of the script (without .pl), the second parameter is the caption
\newcommand{\tcshscript}[2]{
        \begin{itemize}
                \item[]\lstinputlisting[caption=#2,label=#1]{#1.tcsh}
        \end{itemize}
}
\newcommand{\bashscript}[2]{
        \begin{itemize}
                \item[]\lstinputlisting[caption=#2,label=#1]{#1.sh}
        \end{itemize}
}

% Header and footer for when a page split occurs within a problem environment
\newcommand{\enterProblemHeader}[1]{
\nobreak\extramarks{#1}{#1 continued on next page\ldots}\nobreak
\nobreak\extramarks{#1 (continued)}{#1 continued on next page\ldots}\nobreak
}

% Header and footer for when a page split occurs between problem environments
\newcommand{\exitProblemHeader}[1]{
\nobreak\extramarks{#1 (continued)}{#1 continued on next page\ldots}\nobreak
\nobreak\extramarks{#1}{}\nobreak
}

\setcounter{secnumdepth}{0} % Removes default section numbers
\newcounter{moduleProblemCounter} % Creates a counter to keep track of the number of problems

\newcommand{\moduleProblemName}{}
\newenvironment{moduleProblem}[1][Problem \arabic{moduleProblemCounter}]{ % Makes a new environment called moduleProblem which takes 1 argument (custom name) but the default is "Problem #"
\stepcounter{moduleProblemCounter} % Increase counter for number of problems
\renewcommand{\moduleProblemName}{#1} % Assign \moduleProblemName the name of the problem
\section{\moduleProblemName} % Make a section in the document with the custom problem count
\enterProblemHeader{\moduleProblemName} % Header and footer within the environment
}{
\exitProblemHeader{\moduleProblemName} % Header and footer after the environment
}

\newcommand{\problemAnswer}[1]{ % Defines the problem answer command with the content as the only argument
\noindent\framebox[\columnwidth][c]{\begin{minipage}{0.98\columnwidth}#1\end{minipage}} % Makes the box around the problem answer and puts the content inside
}

\newcommand{\moduleSectionName}{}
\newenvironment{moduleSection}[1]{ % New environment for sections within module problems, takes 1 argument - the name of the section
\renewcommand{\moduleSectionName}{#1} % Assign \moduleSectionName to the name of the section from the environment argument
\subsection{\moduleSectionName} % Make a subsection with the custom name of the subsection
\enterProblemHeader{\moduleProblemName\ [\moduleSectionName]} % Header and footer within the environment
}{
\enterProblemHeader{\moduleProblemName} % Header and footer after the environment
}
