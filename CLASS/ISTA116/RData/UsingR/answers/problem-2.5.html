> browsers
          IE      Mozilla    Navigator Unidentified         <NA>

> names(browsers) = c("IE","Mozilla","Navigator","Opera","Un.")
> browsers
       IE   Mozilla Navigator     Opera       Un.
       86         4         5         1         4
> barplot(browsers)
> pie(browsers)
