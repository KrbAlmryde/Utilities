while read file; do

	<${file}.csv cut  -d , -f 1,8-10 | sed -e '1d' -e 's/ /_/g' -e 's/,/ /g' -e 's/True/1/g' -e 's/False/0/g' | awk -v OFS='\t' '$1=$1' | sort > word_${file}.txt
	cat word_${file}.txt #>> word_100s_excel.txt

done <400s.txt
