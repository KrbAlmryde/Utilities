# Script setup

echo "${STROOP}"

mkdir ${STROOP}/ICA

while read subj
do
	for run in voice word
	do

	results= ${STROOP}/ICA
	func= ${STROOP}/${subj}/run*_${run}

		3dTcat -prefix ${func}/${subj}_${run}_tcat ${func}/${subj}_${run}_epan+orig'[4..$]'
		3dAFNItoNIFTI ${func}/${subj}_${run}_tcat+orig ${results}/${subj}_${run}.nii

	done; done < ${STROOP}/Utilities/subjls.txt
